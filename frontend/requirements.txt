# ============================================
# PACKAGE DETAILS
# ============================================

# react-router-dom: Client-side routing for React
# zustand: Lightweight state management (alternative to Redux)
# axios: Promise-based HTTP client
# @supabase/supabase-js: Supabase client for database/auth
# react-hook-form: Performant form library
# yup: Schema validation library
# lucide-react: Beautiful & consistent icon set
# framer-motion: Animation library for React
# @react-three/fiber: React renderer for Three.js (3D graphics)
# three: JavaScript 3D library
# @react-three/drei: Useful helpers for react-three-fiber
# lottie-react: Render Lottie animations
# dayjs: Lightweight date library
# uuid: Generate unique IDs
# prop-types: Runtime type checking for React props
# tailwindcss: Utility-first CSS framework
# postcss: CSS transformation tool
# autoprefixer: PostCSS plugin for vendor prefixes

# ============================================
# BUNDLE SIZE CONSIDERATIONS
# ============================================

# Heavy packages (consider lazy loading):
# - three.js (~600KB)
# - @react-three/fiber
# - @react-three/drei

# Recommended: Load 3D components only when needed using React.lazy()
